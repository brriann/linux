cut and merge files
cut -c[char pos] filename
cut -f[fields] -d[field seperator] filename

cut -d: -f5 /etc/passwd (get all column 5)

cut -d: -f5 /etc/passwd > ~/users_list
cut -d: -f6 /etc/passwd > ~/users_home_list

cat ~/users_list ~/users_home_list (in file order)

paste ~/users_list ~/users_home_list (2 input files are treated as columns or fields in the output file)
